\begin{document}

\usetikzlibrary{
	er,
	arrows,
	positioning
}

\begin{tikzpicture}[
		auto,
		node distance = 20mm,
		manifest/.style = {
			rectangle,
			draw,
			thick,
			inner sep = 0pt,
			minimum width = 15mm,
			minimum height = 10mm
		},
		inv/.style = {
			rectangle,
			draw=none,
			fill=none,
			inner sep = 0pt,
			minimum width = 15mm,
			minimum height = 10mm
		},
		error/.style = {
			ellipse,
			draw,
			thick,
			inner sep = 0pt,
			minimum size = 7mm,
			align = center
		},
		mean/.style={
			regular polygon,
			regular polygon sides = 3,
			draw,
			thick,
			inner sep = 0pt,
			minimum width = 7mm,
			minimum height = 7mm
		},
		path/.style = {
			->,
			thick,
			>=stealth'
		},
		cov/.style = {
			<->,
			thick,
			>=stealth'
		},
	]
	\node[manifest] (X) {$X$};
	\node[manifest] (M) [above right = of X] {$M$};
	\node[manifest] (Y) [below right = of M] {$Y$};
	\node[error] (varepsilonM) [above = of M] {$\varepsilon_{M}$};
	\node[error] (varepsilonY) [above = of Y] {$\varepsilon_{Y}$};

	\draw[path] (X) to node {$c^{\prime}$} (Y);
	\draw[path] (X) to node {$a$} (M);
	\draw[path] (varepsilonM) to node {$1$} (M);
	\draw[path] (varepsilonY) to node {$1$} (Y);
	\draw[cov] (X) to[out = 70, in = 110, looseness = 5] node[above] {$\sigma_{X}^{2}$} (X);
	\draw[cov] (varepsilonM) to[out = -290, in = -250, looseness = 5] node[above] {$\sigma^{2}_{\varepsilon_{M}}$} (varepsilonM);
	\draw[cov] (varepsilonY) to[out = -290, in = -250, looseness = 5] node[above] {$\sigma^{2}_{\varepsilon_{Y}}$} (varepsilonY);
\end{tikzpicture}
