	er,

	positioning
}

\begin{tikzpicture}[
		auto,
		node distance = 20mm,
		manifest/.style = {
			rectangle,
			draw,
			thick,
			inner sep = 0pt,
			minimum width = 15mm,
			minimum height = 10mm
		},

			rectangle,

		},
		error/.style = {
			ellipse,

			inner sep = 0pt,

			align = center

		mean/.style={
			regular polygon,
			regular polygon sides = 3,
			draw,
			thick,

			minimum height = 7mm
		},
		path/.style = {
			->,
			thick,
			>=stealth'
		},
		cov/.style = {

	\node[manifest] (M1) [above right = of X] {$M_{1}$};

	\draw[path] (varepsilonY) to node {$1$} (Y);
	\draw[cov] (X) to[out = 70, in = 110, looseness = 5] node[above] {$\sigma_{X}^{2}$} (X);
	\draw[cov] (varepsilonM1) to[out = -290, in = -250, looseness = 5] node[above] {$\sigma^{2}_{\varepsilon_{M_{1}}}$} (varepsilonM1);
	\draw[cov] (varepsilonM2) to[out = -290, in = -250, looseness = 5] node[above] {$\sigma^{2}_{\varepsilon_{M_{2}}}$} (varepsilonM2);
	\draw[cov] (varepsilonY) to[out = -290, in = -250, looseness = 5] node[above] {$\sigma^{2}_{\varepsilon_{Y}}$} (varepsilonY);
\end{tikzpicture}
